# Terraform Patterns

## Module Structure

```
modules/<name>/
├── main.tf          # resources
├── variables.tf     # input variables
├── outputs.tf       # output values
├── versions.tf      # provider/terraform version constraints
└── README.md        # usage examples (generated by terraform-docs)
```

## Variable Patterns

```hcl
# Simple with validation
variable "environment" {
  type        = string
  description = "Deployment environment"
  validation {
    condition     = contains(["dev", "staging", "prod"], var.environment)
    error_message = "Must be dev, staging, or prod."
  }
}

# Complex object type
variable "node_groups" {
  type = map(object({
    instance_types = list(string)
    min_size       = number
    max_size       = number
    desired_size   = number
    disk_size      = optional(number, 50)
    labels         = optional(map(string), {})
  }))
}

# Sensitive (won't print in plan/apply)
variable "db_password" {
  type      = string
  sensitive = true
}
```

## Resource Patterns

```hcl
# for_each — use over count whenever possible
resource "aws_iam_user" "users" {
  for_each = toset(var.usernames)
  name     = each.key
  tags     = merge(local.common_tags, { Name = each.key })
}

# count — only for simple on/off
resource "aws_cloudwatch_log_group" "optional" {
  count             = var.enable_logging ? 1 : 0
  name              = "/aws/app/logs"
  retention_in_days = 30
}

# Dynamic block
resource "aws_security_group" "this" {
  name   = local.name_prefix
  vpc_id = var.vpc_id

  dynamic "ingress" {
    for_each = var.ingress_rules
    content {
      from_port   = ingress.value.from_port
      to_port     = ingress.value.to_port
      protocol    = ingress.value.protocol
      cidr_blocks = ingress.value.cidr_blocks
      description = lookup(ingress.value, "description", null)
    }
  }

  tags = local.common_tags
}

# Lifecycle rules
resource "aws_instance" "app" {
  ami           = var.ami_id
  instance_type = var.instance_type

  lifecycle {
    create_before_destroy = true    # zero-downtime replacement
    ignore_changes        = [ami]   # managed externally
    prevent_destroy       = true    # protect critical resources
  }
}
```

## Locals Pattern

```hcl
locals {
  name_prefix = "${var.project}-${var.environment}"

  common_tags = {
    Project     = var.project
    Environment = var.environment
    ManagedBy   = "terraform"
    Owner       = var.team
  }

  # Conditional maps
  extra_tags = var.environment == "prod" ? {
    CostCenter = "production"
    Backup     = "required"
  } : {}

  all_tags = merge(local.common_tags, local.extra_tags, var.tags)
}
```

## State Operations Reference

```bash
# Inspect state
terraform state list
terraform state show aws_instance.web
terraform state pull > state.json         # download for inspection

# Move/rename resource (no destroy)
terraform state mv aws_instance.web aws_instance.app
terraform state mv 'module.old' 'module.new'

# Remove from state (without destroying resource)
terraform state rm aws_s3_bucket.orphan

# Import existing resource into state
terraform import aws_s3_bucket.existing my-bucket-name

# Terraform 1.5+ declarative import
import {
  to = aws_s3_bucket.existing
  id = "my-bucket-name"
}

# Unlock stuck state
terraform force-unlock <lock-id>          # verify no active run first!

# Refresh-only plan (detect drift without changes)
terraform plan -refresh-only

# Target single resource (surgical fix)
terraform apply -target=aws_security_group.app

# Replace specific resource (replaces taint)
terraform apply -replace=aws_instance.web
```

## Backend Configuration

```hcl
# S3 backend (AWS)
terraform {
  backend "s3" {
    bucket         = "org-terraform-state"
    key            = "prod/eks/terraform.tfstate"
    region         = "us-east-1"
    encrypt        = true
    kms_key_id     = "alias/terraform-state"
    dynamodb_table = "terraform-state-lock"
  }
}

# GCS backend (GCP)
terraform {
  backend "gcs" {
    bucket = "org-terraform-state"
    prefix = "prod/gke"
  }
}
```

## Multi-Region / Multi-Account Provider Aliases

```hcl
provider "aws" {
  region = "us-east-1"
  alias  = "primary"
}

provider "aws" {
  region = "eu-west-1"
  alias  = "dr"
  assume_role {
    role_arn = "arn:aws:iam::${var.dr_account_id}:role/TerraformRole"
  }
}

resource "aws_s3_bucket" "primary" {
  provider = aws.primary
  bucket   = "${local.name_prefix}-primary"
}

resource "aws_s3_bucket" "dr" {
  provider = aws.dr
  bucket   = "${local.name_prefix}-dr"
}
```

## Terragrunt Patterns

```hcl
# live/prod/eks/terragrunt.hcl
terraform {
  source = "../../../modules//eks"
}

include "root" {
  path = find_in_parent_folders()
}

dependency "vpc" {
  config_path = "../vpc"
  mock_outputs = {
    vpc_id          = "vpc-00000000"
    private_subnets = ["subnet-00000000"]
  }
}

inputs = {
  cluster_name    = "prod-eks"
  vpc_id          = dependency.vpc.outputs.vpc_id
  subnet_ids      = dependency.vpc.outputs.private_subnets
  cluster_version = "1.29"
}
```

## CI/CD Pipeline Pattern

```bash
# Plan (PR validation)
terraform init -backend-config=envs/prod/backend.tfvars
terraform validate
terraform fmt -check -recursive
tfsec .
terraform plan -var-file=envs/prod/terraform.tfvars -out=tfplan

# Apply (after merge to main)
terraform apply tfplan
```

## Common Anti-Patterns

| Anti-Pattern | Fix |
|-------------|-----|
| `count` with list (order shifts = destroy/recreate) | `for_each` with map/set |
| No state backend (local tfstate) | S3/GCS/TFC remote backend |
| No state locking | DynamoDB/GCS locking |
| Not committing `.terraform.lock.hcl` | Always commit lockfile |
| Hardcoded region/account IDs | Variables + data sources |
| Monolithic root with 200+ resources | Split into modules |
| `terraform apply -auto-approve` in prod CI | Plan review gate |
| No `validation` blocks on variables | Add constraints |
| Secrets in `.tfvars` files | Secrets Manager + data source |
